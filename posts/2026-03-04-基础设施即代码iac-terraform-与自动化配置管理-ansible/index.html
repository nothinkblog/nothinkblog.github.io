  required_providers {

      version = "2.21.0" # 建议指定版本
    }
  }
}

编写 Ansible Playbook 初始化系统

  • Terraform 中自动生成 Inventory
# 自动生成 Ansible 的 hosts 文件
resource "local_file" "ansible_inventory" {
  content  = <<EOT
[vultr_servers]
${vultr_instance.my_server.v6_main_ip} ansible_user=root
EOT
  filename = "inventory.ini"
}
  • init_server.yml 代码如下
---
- name: 初始化 Vultr VPS
  hosts: vultr_servers
  gather_facts: yes
  tasks:
    - name: 更新 apt 缓存并升级系统
      apt:
        update_cache: yes
        upgrade: dist

    - name: 安装常用工具 (htop, ufw)
      apt:
        name:
          - htop
          - ufw
        state: present

    - name: 配置 UFW 允许 SSH
      ufw:
        rule: allow
        name: SSH

    - name: 开启 UFW 防火墙
      ufw:
        state: enabled
  • 运行并初始化
ansible-playbook -i inventory.ini init_server.yml

Terraform 直接触发 Ansible

  • 如果想在 terraform apply 完成的那一刻自动触发初始化，可以在 vultr_instance 资源里加入 local-exec
resource "vultr_instance" "my_server" {
  # ... 原有配置 ...

  provisioner "local-exec" {
    command = "sleep 30 && ansible-playbook -i inventory.ini init_server.yml"
  }
}

彻底销毁所有资源

terraform destroy
